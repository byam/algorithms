        // binary search
        auto it = lower_bound(all(a), b);
        int ans = 0;
        // not found
        if (it == a.end()) {
            ans = abs(b - a[n - 1]);
        } else {
            // get index
            int idx = it - a.begin();
            ans = abs(b - a[idx]);

    ll rd(a, b, c);

    ll ans = INT_MAX;
    ll ma = 10000;

    for (ll i = 0; i < ma; i++) {
        for (ll j = 0; j < ma - i; j++) {
            // 工夫する

        // check
        bool ok = true;
        for (int i = 0; i < n - 1; i++) {

            if (e.find({cur, next}) != e.end()) ok = false;

        if (ok) {
            int time = 0;
            for (int i = 0; i < n; i++) {

            }
            chmin(ans, time);
        }

## 064. 段差を考えよう

- Problem

  - ![image](https://raw.githubusercontent.com/E869120/kyopro_educational_90/main/editorial/064.jpg)
  - [cpp](https://github.com/E869120/kyopro_educational_90/blob/main/sol/064.cpp)

<details>
  <summary> Code </summary>

```cpp
    // in
    int rd(n, q);
    vll rdv(a, n);

    // each diffs
    vll b(n);
    ll sum = 0;
    for (int i = 0; i < n - 1; i++) {
        b[i] = a[i + 1] - a[i];
        sum += abs(b[i]);
    }

    // queries
    for (int i = 0; i < q; i++) {
        ll rd(l, r, v);
        l--;
        r--;

        ll mae = abs(b[l - 1]) + abs(b[r]);

        // b(l - 1) + v
        if (l >= 1) {
            b[l - 1] += v;
        }

        // b(r) - v
        if (r <= n - 2) {
            b[r] -= v;
        }

        // sum
        ll ato = abs(b[l - 1]) + abs(b[r]);
        sum += ato - mae;
        out(sum);
    }
```

</details>
